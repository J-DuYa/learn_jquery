```javascript
var access = jQuery.access = function (elems, fn, key, value, chainable, emptyGet, raw) {
	var len = elems.length,
		bulk = key == null;

	// 设置多个值
	if (jQuery.type(key) === "object") {

	// value存在，只设置一个值
	} else if (value !== undefined) {
		chainable = true;

		if (bulk) {
			if (raw) {
				// set时，fn只接受一个参数
				fn.call(elems, value);
				fn = null;
			}
		}
	}

	// set时chainable为true
	return chainable ?
		// set 返回jQuery对象
		elems : 

		// get
		bulk ? 
			// 通过fn返回get的值
			fn.call(elems) :
			len ? fn(elems[0], key) : emptyGet;

};
```
