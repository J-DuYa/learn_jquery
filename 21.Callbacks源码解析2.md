构建最基本的模型来模拟add、fire方法

```javascript
jQuery.Callbacks = function (options) {
	var list = [],

		self = {
			add: function (fn) {
				list.push(fn);
			},

			fire: function (args) {
				list.forEach(function (fn) {
					fn(args);
				});
			}
		};

	return self;
};
```
验证，结果和jQuery完全一致

```javascript
var callbacks = $.Callbacks();
function fn1(val) {
    console.log("fn1 says:" + val);
}
function fn2(val) {
    console.log("fn2 says:" + val);
}

callbacks.add(fn1);
callbacks.fire("foo");
console.log("..............")
callbacks.add(fn2);
callbacks.fire("bar");

// 输出
// fn1 says:foo
// ..............
// fn1 says:bar
// fn2 says:bar
```
