#### jQuery迭代器

###### jQuery源码

```javascript
jQuery.extend({
	each: function (obj, callback, args) {
		var i = 0,
			length = obj.length,
			isArray = Array.isArray(obj);

		if (isArray) {
			for (; i < length; i++) {
				//callback(索引, 值)
				callback.call(obj[i], i, obj[i]);
			}
		} else {
			for (i in obj) {
				//callback(属性, 值)
				callback.call(obj[i], i, obj[i]);
			}
		}

		return obj;
	}
});
```

###### 静态调用

```javascript
//遍历数组
$.each([1, 2, 3], function (index, value) {
	console.log(index + ' ' + value);
});

//输出
//0 1
//1 2
//2 3

//遍历对象
$.each({a:'aa',b:'bb',c:'cc'}, function (index, value) {
	console.log(index + ' ' + value);
});

//输出
//a aa
//b bb
//c cc
```
